# stop app so we can recreate the database cleanly
docker compose stop app

# drop the app DB and recreate it EMPTY (FORCE closes any lingering connections)
docker exec -i socialvista-db-1 sh -c \
  'psql -U "$POSTGRES_USER" -d postgres -v ON_ERROR_STOP=1 \
     -c "DROP DATABASE IF EXISTS \"$POSTGRES_DB\" WITH (FORCE);" \
     -c "CREATE DATABASE \"$POSTGRES_DB\" OWNER \"$POSTGRES_USER\";"'

# plain full dump from Neon (no --clean) → restore into the empty DB
docker exec -e PGURL="$NEON_URL" socialvista-db-1 \
  sh -c 'pg_dump "$PGURL" --no-owner --no-acl' \
  | docker exec -i socialvista-db-1 sh -c 'psql -U "$POSTGRES_USER" -d "$POSTGRES_DB" -v ON_ERROR_STOP=1'

# restart app + clear the URL
docker compose start app
unset NEON_URL
