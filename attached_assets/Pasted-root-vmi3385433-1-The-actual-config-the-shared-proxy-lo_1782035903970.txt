# Shared Caddy reverse proxy, run BEHIND Cloudflare (SSL/TLS mode: Full (strict)).
# Each site serves a Cloudflare Origin Certificate (trusted by Cloudflare's edge),
# so the Cloudflare<->server hop is encrypted end-to-end. Caddy does NOT use
# Let's Encrypt here — the explicit `tls` directive pins the origin cert and
# disables ACME (which can't complete while traffic is proxied through Cloudflare).
#
# Caddy routes by the request's Host header, so one server IP serves many domains:
# each project's `web` container is reachable on the shared `web` network under the
# alias referenced below. Drop each domain's origin cert + key into ./certs/<name>/
# (mounted read-only at /etc/caddy/certs — gitignored, never committed).
#
# RECOMMENDED HARDENING (Authenticated Origin Pulls): lock the origin so ONLY
# Cloudflare can connect. Otherwise anyone who finds your server IP can hit the
# origin directly and forge X-Forwarded-For (defeating per-IP rate limiting, which
# trusts the proxy chain via TRUST_PROXY). To enable, see deploy/README.md
# "Step 5 — Harden": turn on Authenticated Origin Pulls in Cloudflare, download the
# origin-pull CA to ./certs/cloudflare/origin-pull-ca.pem, and switch each site's
# single-line `tls ...` to the commented block form shown below it.

{$AGENTOPIA_DOMAIN} {
        tls /etc/caddy/certs/agentopia/origin.pem /etc/caddy/certs/agentopia/origin.key
        # Hardened (Authenticated Origin Pulls) — replace the `tls` line above with:
        # tls /etc/caddy/certs/agentopia/origin.pem /etc/caddy/certs/agentopia/origin.key {
        #       client_auth {
        #               mode require_and_verify
        #               trust_pool file /etc/caddy/certs/cloudflare/origin-pull-ca.pem
        #       }
        # }
        reverse_proxy agentopia-web:80
}

services:
  # Express API (bundled to dist/index.mjs). Internal only — Caddy proxies to it.
  api:
    build:
      context: .
      target: api
    image: polypay-api
    restart: unless-stopped
    env_file: .env
    environment:
      NODE_ENV: production
      PORT: "8080"
    expose:
      - "8080"
    healthcheck:
      test: ["CMD", "curl", "-fsS", "http://localhost:8080/api/healthz"]
      interval: 30s
      timeout: 5s
      retries: 5
      start_period: 20s
    networks:
      - polypay

  # Caddy: serves the SPA at / and reverse-proxies /api -> api. Public entrypoint.
  caddy:
    build:
      context: .
      target: web
    image: polypay-web
    restart: unless-stopped
    depends_on:
      - api
    # Only the HTTPS port is published. Cloudflare ("Full (strict)") connects to
    # the origin over HTTPS, so host port 80 is unnecessary and is intentionally
    # not published — that avoids colliding with any other service on this host.
    # Override CADDY_HTTPS_PORT in .env (e.g. 8443) to run alongside another
    # program that already owns 443; then add a Cloudflare Origin Rule rewriting
    # the origin port for polypay.in. Leave unset to use the standard 443.
    ports:
      - "${CADDY_HTTPS_PORT:-443}:443"
    volumes:
      - ./certs:/etc/caddy/certs:ro
      - caddy_data:/data
      - caddy_config:/config
    networks:
      - polypay

  # One-off helper: push the Drizzle schema to the Neon production DB.
  # Does NOT start with `up`. Run it manually:
  #   docker compose --profile tools run --rm migrate
  migrate:
    profiles: ["tools"]
    build:
      context: .
      target: deps
    image: polypay-deps
    env_file: .env
    # `$$` escapes compose interpolation so the container shell expands the var
    # that env_file injected. Use run:push-force for non-interactive applies.
    command: sh -c 'DATABASE_URL="$$NEON_DATABASE_URL" pnpm --filter @workspace/db run push'
    networks:
      - polypay
